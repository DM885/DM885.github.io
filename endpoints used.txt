POST /run -- Adds a run to the DB
BODY:
    {
        model: Number,
        dataset: Number,
        solvers: [{
            solver: String,
            flagA: Boolean,
            flagF: Boolean,
            flagP: Number,
        }],
    }
RESPONSE:
    {
        error: Boolean
    }

MODEL:

GET /model - Gets all models-metadata
[{
    name: string,
    id: number,
}]


GET /model/:id - Gets the model
{
    name: string,
    content: string,
    id: number,
}


POST /model - Adds the given model
{
    name: string,
    content: string,
}
RESPONSE:
{
    error: Boolean
}


PUT /model/:id - Updates the given model
{
    name: string,
    content: string,
}
RESPONSE:
{
    error: Boolean
}
