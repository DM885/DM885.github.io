POST /run -- Adds a run to the DB
BODY:
    {
        model: Number,
        dataset: Number,
        solvers: [{
            solver: String,
            flagA: Boolean,
            flagF: Boolean,
            flagP: Number,
        }],
    }
RESPONSE:
    {
        error: Boolean
    }

GET /model - Gets all models-metadata
GET /model/:id - Gets the model

GET /data -- Gets all data-metadata
GET /data/:id -- Gets the data
